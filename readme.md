# CLog

```ini
[clog]
msgsize=1024 # [message size]:number; default 1024
datetime=true # [show datetime]:true,false; default true
loglevel=true # [show log level]:true,false; default true
function=true # [show function name]:true,false; default false
position=true # [show log position]:true,false; default false
stdout=true # [print log on stdout]:true,false; default false
level=WARN # [current log level]:ERROR,WARN,INFO,DEBUG; default WARN
dir=E:\Projects\clog # [log file directory]:directory string; default current dir or /tmp/
name=clogtest # [log file name]:name string; default clog
```
